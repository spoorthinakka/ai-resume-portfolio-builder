# === AI Resume & Portfolio Builder – Requirements ===

# Install all dependencies automatically with:
#     pip install -r requirements.txt
#
# If installing individually:
#     pip install streamlit
#     pip install huggingface_hub
#     pip install python-docx
#     pip install fpdf
#     pip install python-dotenv
#
# These libraries enable:
#  • Streamlit web interface
#  • AI text generation via Hugging Face API
#  • Resume export in PDF & DOCX formats
#  • Secure API key handling using .env file

streamlit         # Web app framework for interactive UI
huggingface_hub   # Connects to Hugging Face Inference API
python-docx       # Generates .docx (Word) resumes
fpdf              # Creates downloadable PDF resumes
python-dotenv     # Loads environment variables from .env file
